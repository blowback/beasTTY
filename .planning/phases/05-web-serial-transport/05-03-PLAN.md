Read loop shape (D-35 + 05-RESEARCH Pattern 2 lines 390-419):
```js
async function runReadLoop(port) {
    while (port.readable) {                  // OUTER — re-enters on non-fatal errors
        reader = port.readable.getReader();
        try {
            while (true) {                   // INNER — cancellable via reader.cancel()
                const { value, done } = await reader.read();
                if (done) break;             // D-36 — cancel resolves with done:true
                term.feed(value);            // Phase 2 feed_silent; no TextDecoder
                sampleBell();                // Phase 3 post-feed invariant
                drainHostReply('serial');    // Phase 2 host-reply accessor drain
                requestFrame();              // Phase 3 dirty-repaint wake
            }
        } catch (err) {
            handleReadError(err);
        } finally {
            try { reader.releaseLock(); } catch {}
            reader = null;
        }
    }
    // port.readable is null → port is dead. Wave 4 handles the reconnect path.
    try { await port.close(); } catch {}
    setState('disconnected');
}
```
