// ===== processFiles — runs validation + truncation + modal flow =====
async function processFiles(filesArr) {
    // Build per-file rows: { kind: 'rewrite' | 'unchanged' | 'rejected',
    //                        original, rewritten?, reason?, bytes? }
    const rows = [];
    const surviving = [];
    for (const f of filesArr) {
        const original = f.name;
        const validation = validateCpmFilename(original);
        if (!validation.ok) {
            rows.push({ kind: 'rejected', original, reason: validation.reason });
            continue;
        }
        const rewritten = truncateCpm83(original);
        const ab = await f.arrayBuffer();
        const bytes = new Uint8Array(ab);
        if (rewritten === original) {
            rows.push({ kind: 'unchanged', original });
        } else {
            rows.push({ kind: 'rewrite', original, rewritten });
        }
        surviving.push({ name: rewritten, bytes });
    }

    // Show modal; await user choice.
    const userConfirmed = await showConfirmModal(rows, surviving);
    if (!userConfirmed) return;

    // Hand off to transport/slide.js.
    if (enterSendModeFn) {
        enterSendModeFn({ files: surviving });
    }
}

/**
 * Validate a filename against the CP/M-invalid character set (D-06).
 *
 * Returns { ok: true, reason: null } if valid, otherwise an object with
 * a human-readable reason string suitable for the modal rejection row.
 *
 * Rules:
 *   - empty string → invalid
 *   - leading dot (dotfile) → invalid
 *   - control characters (codepoint < 0x20) → invalid
 *   - non-ASCII (codepoint >= 0x80) → invalid
 *   - any char in <,>,,,;,:,=,?,*,[,] → invalid
 *
 * The bytes-≥-0x80 check uses charCodeAt (UTF-16 code unit). For BMP
 * codepoints this matches; for surrogate pairs the high surrogate is
 * always ≥ 0xD800 ≥ 0x80 so it triggers rejection on the first half
 * (correct outcome).
 */
export function validateCpmFilename(name) {
    if (!name || name.length === 0) return { ok: false, reason: 'empty filename' };
    if (name.startsWith('.')) return { ok: false, reason: 'leading-dot dotfile' };
    for (let i = 0; i < name.length; i++) {
        const c = name.charCodeAt(i);
        if (c < 0x20) {
            return { ok: false, reason: `control character 0x${c.toString(16).padStart(2, '0')}` };
        }
        if (c >= 0x80) {
            return { ok: false, reason: `non-ASCII byte 0x${c.toString(16).padStart(2, '0')}` };
        }
        const ch = name[i];
        if (CPM_INVALID_CHARS.has(ch)) {
            return { ok: false, reason: `invalid CP/M character '${ch}'` };
        }
    }
    return { ok: true, reason: null };
}

/**
 * Apply the CP/M 8.3 truncation algorithm (D-07).
 *
 * - Uppercase via String#toUpperCase()
 * - Split on the FINAL `.`; truncate base to 8, ext to 3
 * - No extension → truncate base to 8, no dot in result
 * - Multi-dot files split on last dot (e.g. `my.tar.gz` → base=`my.tar` → 6 chars,
 *   ext=`gz` → 2 chars → result `MY.TAR.GZ`)
 *
 * Caller is expected to have already passed validateCpmFilename — this
 * function does NOT re-validate (e.g., it does not reject leading-dot files).
 */
export function truncateCpm83(name) {
    const upper = name.toUpperCase();
    const lastDot = upper.lastIndexOf('.');
    if (lastDot < 0) {
        return upper.slice(0, 8);
    }
    const base = upper.slice(0, lastDot).slice(0, 8);
    const ext = upper.slice(lastDot + 1).slice(0, 3);
    return ext.length > 0 ? `${base}.${ext}` : base;
}

/**
 * Pack file metadata per CONTEXT D-09:
 *
 *   <u32 LE file_count>
 *   for each file:
 *     <u32 LE name_len>
 *     <name bytes (UTF-8 / ASCII; already CP/M-validated + truncated)>
 *     <u32 LE size>
 *
 * Returns Uint8Array. Caller passes [{ name: string, bytes: Uint8Array }, ...].
 */
export function packSendMetadata(files) {
    const enc = new TextEncoder();
    const nameBytesArr = files.map((f) => enc.encode(f.name));
    let totalLen = 4;
    for (const nb of nameBytesArr) {
        totalLen += 4 + nb.length + 4;
    }
    const buf = new Uint8Array(totalLen);
    const dv = new DataView(buf.buffer);
    let cursor = 0;
    dv.setUint32(cursor, files.length, true /* LE */); cursor += 4;
    for (let i = 0; i < files.length; i++) {
        const nb = nameBytesArr[i];
        dv.setUint32(cursor, nb.length, true); cursor += 4;
        buf.set(nb, cursor); cursor += nb.length;
        dv.setUint32(cursor, files[i].bytes.length, true); cursor += 4;
    }
    return buf;
}

## Trust Boundaries (Plan 09-03)

| Boundary | Description |
|----------|-------------|
| OS → Browser (drag-drop / file picker) | User-supplied files cross from OS into the browser via standard File API. Files contain user-controlled bytes + filenames. |
| File-source.js (validation + truncation + pack) → transport/slide.js (enterSendMode) | Validated + truncated filenames + raw bytes are packed into D-09 metadata blob and handed to enterSendMode. The validation is the trust boundary. |
| `<dialog>` modal user input → enterSendMode | User clicks Send → calls enterSendMode. Click is a user gesture; no programmatic invocation by malicious script can synthesize a real "send" without the user's interaction. |

## STRIDE Threat Register (Plan 09-03 — UI/file-source scope)

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-03-01 | Tampering | filename injection — user drops a file with shell metacharacters in its name (e.g., `bad;rm -rf /.txt`) hoping it ends up in the auto-typed CP/M command line | mitigate (closed by structure) | The auto-typed command (Plan 09-02) is the static literal `B:SLIDE R\r` ONLY; filenames travel via the SLIDE protocol HEADER FRAME (build_header_frame) AFTER handshake — they NEVER concatenate into a CP/M command line. validateCpmFilename additionally rejects `;`, `|`, etc. via the CP/M-invalid set + control-character check (D-06). DEFENSE-IN-DEPTH AT TWO LAYERS: (1) auto-typed command structurally separate from filenames, (2) CP/M validation rejects shell metacharacters even for the SLIDE protocol filename. |
| T-09-03-02 | Tampering | CP/M-invalid character bypass — malformed file (null byte in filename, UTF-8 normalization edge case) reaches Rust SM with bad bytes | mitigate | validateCpmFilename rejects: control characters (codepoint < 0x20, INCLUDING null byte 0x00), non-ASCII (codepoint >= 0x80, INCLUDING UTF-8 multi-byte sequences which cause the high bytes to be ≥ 0x80). Rejection happens BEFORE the modal is shown (validateCpmFilename runs in processFiles before buildModalRows). The modal lists rejected files but offers no path to send them — the surviving array is reduced. enterSendMode receives ONLY validated names. CONTEXT D-06+D-07 lock this; the JS-side validator is the single gate per CLAUDE.md architecture (Rust trusts JS). |
| T-09-03-03 | Tampering | drag-drop event spoofing — malicious page-embedded iframe or non-File MIME data tricks the dragenter handler | mitigate | `dataTransfer.types.includes('Files')` (D-04 verbatim) returns true ONLY for OS file drags, not for text/URL/HTML/JSON/iframe drags. The check runs at dragenter / dragover / dragleave / drop — all four events guard on isFileDrag. A hostile iframe attempting `dataTransfer.setData('Files', ...)` cannot fake the 'Files' type because that type is reserved for OS-originated file drags by the HTML Living Standard. Verified by inspection of dataTransfer.types semantics. |
| T-09-03-04 | Information Disclosure | `<dialog>` modal escape — modal-open + page-level keyboard listeners collide if Esc handling isn't routed through the dialog's native cancel event | mitigate | Native `<dialog>` Esc-cancel is built-in browser behavior — the Esc key fires `cancel` event on the dialog and closes via dialog.close(''). No page-level Esc listener is added in Phase 9. Plan 09-04 Playwright test asserts Esc closes the modal as cancel. The Phase 10 Esc-cancel-disambiguation chain (slot 2 of 4) will be built ON TOP of this — not against it — because the chain only fires when the dialog is closed (no top-bar modal active). |
| T-09-03-05 | Tampering | drop event during active SLIDE session corrupts wire | mitigate | onDrop / onDragEnter / onDragOver / onDragLeave all check `isSessionActive()` BEFORE preventDefault and BEFORE setting [data-drop-target]. During an active session, drops are silently consumed (no overlay, no modal, no enterSendMode call). UI-SPEC §"Drag-drop during active SLIDE session" locks this behavior — Phase 11 SLIDE-11 adds the user-visible "Transfer in progress — cancel first" chip. |
| T-09-03-06 | Denial of Service | user drops 10,000 files at once, modal lists overflow | accept | The `<ul id="send-modal-list">` has `overflow-y: auto` + `flex: 1 1 auto; min-height: 0` so the inner list scrolls within the dialog (UI-SPEC §Modal long-list overflow). Browser handles 10K `<li>` elements without DOM stress. The modal title says "Sending 10000 files via SLIDE" — accurate. The user can click Don't send to back out. No mitigation needed beyond the existing scroll. |
| T-09-03-07 | Tampering | malicious script overrides `enterSendMode` import in main.js to inject arbitrary metadata | accept | ES module bindings are read-only by spec; an attacker would need to replace the source file on disk (out of browser-script scope). Defense is at OS / build-toolchain layer, not application layer. |
