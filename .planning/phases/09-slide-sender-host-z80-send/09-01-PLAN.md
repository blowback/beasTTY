```rust
pub fn build_frame(seq: u8, payload: &[u8]) -> Vec<u8> {
    let length = payload.len();
    let mut crc_data = vec![seq, (length >> 8) as u8, (length & 0xFF) as u8];
    crc_data.extend_from_slice(payload);
    let crc = crc16_ccitt(&crc_data);
    let mut frame = vec![SOF, seq, (length >> 8) as u8, (length & 0xFF) as u8];
    frame.extend_from_slice(payload);
    frame.push((crc >> 8) as u8);
    frame.push((crc & 0xFF) as u8);
    frame
}
```

      ```rust
      /// [SOF][SEQ][LEN_HI][LEN_LO][PAYLOAD][CRC_HI][CRC_LO]. Pushes

      /// CRC scope: [seq, len_hi, len_lo, ...payload] (NOT including SOF

      /// framer.rs module doc.
      pub fn build_frame_into(buf: &mut Vec<u8>, seq: u8, payload: &[u8]) {
          let length = payload.len();
          let len_hi = (length >> 8) as u8;
          let len_lo = (length & 0xFF) as u8;
          // CRC over [seq, len_hi, len_lo, ...payload]
          let mut crc_buf = Vec::with_capacity(3 + length);
          crc_buf.push(seq); crc_buf.push(len_hi); crc_buf.push(len_lo);
      ```

```rust
      }

fn pseudo_random_bytes(len: usize) -> Vec<u8> {
    // Deterministic pseudo-random — xorshift32 seeded with fixed seed so the
    // test is byte-stable across runs.
    let mut out = Vec::with_capacity(len);
    let mut state: u32 = 0xdeadbeef;
    for _ in 0..len {
        state ^= state << 13;
        state ^= state >> 17;
        state ^= state << 5;
        out.push((state & 0xFF) as u8);
    }
    out
}

#[derive(Clone, Copy)]
enum BotInjectMode {
    None,
    NakOnFirstFrame,    // emits NAK(1) once, then resumes ACKing normally
    CanMidStream,        // emits CTRL_CAN after first data frame is seen
}

struct MockReceiver {
    mode: BotInjectMode,
    // Reassembly: per-file received bytes (key = file_idx, value = payload bytes).
    received_files: Vec<Vec<u8>>,
    received_filenames: Vec<Vec<u8>>,
    // Internal frame parser state for incoming sender frames.
    parse_buf: Vec<u8>,
    cursor: usize,
    // Track whether NAK has already been injected (one-shot).
    nak_already_injected: bool,
    can_already_injected: bool,
    // FIN observed?
    fin_observed: bool,
    // Whether bot has emitted RDY yet.
    rdy_emitted: bool,
}

impl MockReceiver {
    fn new(mode: BotInjectMode) -> Self {
        Self {
            mode,
            received_files: Vec::new(),
            received_filenames: Vec::new(),
            parse_buf: Vec::with_capacity(8192),
            cursor: 0,
            nak_already_injected: false,
            can_already_injected: false,
            fin_observed: false,
            rdy_emitted: false,
        }
    }

    /// Feed sender outbound bytes; return bot's response bytes (ACK/NAK/CAN/FIN).
    fn feed_sender_bytes(&mut self, sender_bytes: &[u8]) -> Vec<u8> {
        self.parse_buf.extend_from_slice(sender_bytes);
        let mut response = Vec::new();

        // Emit RDY echo on first byte the sender has produced (sender pushed CTRL_RDY first).
        if !self.rdy_emitted && !self.parse_buf.is_empty() {
            // Sender's first push is CTRL_RDY (0x11) — strip it from parse_buf
            // since it's a control byte, not a frame.
            if self.parse_buf[0] == 0x11 /* CTRL_RDY */ {
                self.parse_buf.drain(0..1);
            }
            response.push(0x11);  // CTRL_RDY echo
            self.rdy_emitted = true;
        }

        // Parse complete frames from parse_buf.
        loop {
            // Look for next frame boundary or control byte.
            if self.cursor >= self.parse_buf.len() { break; }
            let b = self.parse_buf[self.cursor];
            match b {
                0x01 => {
                    // SOF — try to parse a complete frame.
                    if self.parse_buf.len() < self.cursor + 4 { break; }  // need SOF + SEQ + LEN_HI + LEN_LO
                    let seq = self.parse_buf[self.cursor + 1];
                    let len_hi = self.parse_buf[self.cursor + 2];
                    let len_lo = self.parse_buf[self.cursor + 3];
                    let payload_len = ((len_hi as usize) << 8) | (len_lo as usize);
                    let frame_total = 4 + payload_len + 2;  // SOF + SEQ + LEN_H + LEN_L + payload + CRC_H + CRC_L
                    if self.parse_buf.len() < self.cursor + frame_total { break; }
                    let payload = &self.parse_buf[self.cursor + 4 .. self.cursor + 4 + payload_len].to_vec();
                    let frame_bytes = &self.parse_buf[self.cursor .. self.cursor + frame_total];
                    self.handle_frame(seq, payload, frame_bytes, &mut response);
                    self.cursor += frame_total;
                }
                0x04 /* CTRL_FIN */ => {
                    self.fin_observed = true;
                    response.push(0x04);  // FIN echo
                    self.cursor += 1;
                }
                0x18 /* CTRL_CAN */ => {
                    // sender CAN echo (D-19) — bot doesn't need to re-echo.
                    self.cursor += 1;
                }
                _ => {
                    // Skip unknown bytes (control-byte ACK aux mid-stream from
                    // the sender wouldn't reach the bot in the production wire,
                    // but defensively skip).
                    self.cursor += 1;
                }
            }
        }
        // Compact parse_buf: drain consumed prefix.
        self.parse_buf.drain(0..self.cursor);
        self.cursor = 0;
        response
    }

    fn handle_frame(&mut self, seq: u8, payload: &[u8], _full_frame: &[u8], response: &mut Vec<u8>) {
        if seq == 0 {
            // Header frame: payload = name + null + size_le_u32.
            let null_pos = payload.iter().position(|&b| b == 0).expect("header has null");
            let name = payload[..null_pos].to_vec();
            self.received_filenames.push(name);
            self.received_files.push(Vec::new());
            response.push(0x06); response.push(0); // CTRL_ACK + seq=0
        } else {
            // Data frame.
            // Inject CAN before processing if requested.
            if matches!(self.mode, BotInjectMode::CanMidStream) && !self.can_already_injected {
                response.push(0x18);  // CTRL_CAN
                self.can_already_injected = true;
                return;
            }
            // Inject NAK on first data frame (seq=1) if requested.
            if matches!(self.mode, BotInjectMode::NakOnFirstFrame) && !self.nak_already_injected && seq == 1 {
                response.push(0x15); response.push(seq);  // CTRL_NAK + seq=1
                self.nak_already_injected = true;
                return;
            }
            if payload.is_empty() {
                // EOF marker — ACK the EOF seq.
                response.push(0x06); response.push(seq);
            } else {
                // Append payload to current file.
                let cur = self.received_files.last_mut().expect("header seen first");
                cur.extend_from_slice(payload);
                response.push(0x06); response.push(seq);
            }
        }
    }

    fn received_bytes(&self, file_idx: usize) -> &[u8] {
        &self.received_files[file_idx]
    }

    fn fin_observed(&self) -> bool {
        self.fin_observed
    }
}

/// Drive a sender Slide through its lifecycle against `bot` until the sender
/// reaches Done or Error or CancelPending. Returns at most `max_iter`
/// iterations to avoid infinite loops on test failure.
fn drive_session(sender: &mut Slide, bot: &mut MockReceiver, file_payloads: &[Vec<u8>], max_iter: usize) {
    let mut payload_offsets = vec![0usize; file_payloads.len()];

    for _ in 0..max_iter {
        // Drain sender outbound to the bot.
        let out = outbound_snapshot(sender);
        if !out.is_empty() {
            sender.clear_outbound();
            let response = bot.feed_sender_bytes(&out);
            if !response.is_empty() {
                sender.feed_chunk(&response);
            }
        }

        // Drain sender events; on EVT_ACK in DataPhase that doesn't advance to
        // FILE_COMPLETE, JS would push next data chunk. We mimic that here.
        let mut want_next_chunk = false;
        loop {
            let e = sender.take_event_packed();
            if e == 0 { break; }
            let kind = e & 0xFFFF_0000;
            if kind == EVT_ACK {
                want_next_chunk = true;
            }
            if kind == EVT_RETRANSMIT_NEEDED {
                // Re-feed the payload chunk for the requested seq.
                // Simplification: the test drives a single chunk per file;
                // on NAK retransmit we re-feed that chunk.
                let aux = (e & 0xFFFF) as u8;
                // For aux == 1 (first data frame), find current file's payload[0..len].
                // Use ctx via state == DataPhase semantics.
                let _ = aux;
                want_next_chunk = true;
            }
            if kind == EVT_SESSION_COMPLETE { return; }
        }

        // If sender is in DataPhase, push next chunk if there's more payload.
        if sender.state() == SlideState::DataPhase as u32 {
            // Determine current file index by inspecting events emitted earlier.
            // Simplification: walk file_payloads; the test sets up chunks-per-file
            // such that the `eof=true` path covers the full payload in one chunk.
            // Find the first file with offset < len.
            let cur_idx = payload_offsets.iter()
                .position(|&off| off < file_payloads.iter().nth(0).map(|p| p.len()).unwrap_or(0));
            if let Some(idx) = cur_idx {
                let remaining = &file_payloads[idx][payload_offsets[idx]..];
                let chunk_len = remaining.len().min(FRAME_SIZE);
                let is_eof = chunk_len == remaining.len();
                let chunk: Vec<u8> = remaining[..chunk_len].to_vec();
                sender.feed_send_chunk(&chunk, is_eof);
                payload_offsets[idx] += chunk_len;
            }
        } else if want_next_chunk {
            // No-op — sender is between phases.
        }

        // Termination conditions.
        let st = sender.state();
        if st == SlideState::Done as u32 { return; }
        if st == SlideState::Error as u32 { return; }
        if st == SlideState::CancelPending as u32 { return; }

        if outbound_snapshot(sender).is_empty() && sender.outbound_len() == 0 {
            // Nothing to drive forward.
            // Avoid spinning: break only if sender has truly stalled.
            // The test should drive the bot one more time.
        }
    }
    panic!("drive_session reached max_iter without terminating; sender state = {}", sender.state());
}
```

| T-09-01-01 | Tampering | metadata blob from JS — malformed `file_count` or `name_len` could cause OOB read inside `enter_send_mode` parsing | accept | Phase 9 trusts JS validation per CONTEXT D-09. Rust SM panics on malformed input (debug_assert + array indexing); the panic is a wasm trap that JS catches as a thrown `RuntimeError`. | T-09-01-02 | Information Disclosure | `outbound_ptr` exposes Rust linear memory to JS; if `OUTBOUND_RESERVE` is too small, a Vec reallocation invalidates the cached JS view → JS could then read past valid data into adjacent allocations | mitigate | OUTBOUND_RESERVE = 4128 (Task 2) sized for 4 max-size frames + slack. The new test `outbound_ptr_stable_across_sender_window_pushes` proves no reallocation occurs across 4 max-size frames. The mirror `OUTBOUND_VIEW_CAP` constant in `transport/slide.js` MUST grow in lockstep (Plan 09-02) — else JS view caps at 16 bytes regardless of reserve. |
| T-09-01-03 | Denial of Service | inbound stream of CTRL_NAK forces sender SM into infinite retransmit loop (slide-rs/send.rs has no upper bound) | accept | Per CONTEXT Claude's Discretion §"Sender retry budget", Phase 9 matches slide-rs (no upper bound). Production wire is Web Serial USB; a hostile/buggy Z80 could trigger this but the user can always force-quit the tab. SEND_NAK_BUDGET introduction is Phase 12 hardening (deferred). |
| T-09-01-04 | Denial of Service | malformed inbound CTRL_FIN sequence stalls sender in FinPending forever | accept | No timeout in Rust core (no `std::time` per ADR-003). JS can call `slide.cancel()` or `slide.force_idle()` (Phase 7 D-06/D-07) to escape. Phase 11 wires the user-visible Cancel UI. |
| T-09-01-05 | Tampering | inbound EVT_CAN from any sender state echoes CTRL_CAN → CancelPending; an attacker who controls inbound bytes can abort sessions | mitigate | This IS the protocol (ADR-003 D-05 strict bidirectional). The threat is the protocol's intended behaviour; the only mitigation is the wire-level integrity provided by USB-serial physical access. |
| T-09-01-06 | Tampering | inbound NAK with arbitrary `seq` causes sender to emit `EVT_RETRANSMIT_NEEDED | seq` into events ring; if JS naively accesses `payload[seq * FRAME_SIZE..]`, a malicious seq value could cause OOB index in JS | accept | The seq value is `u8` (0..255); JS-side payload subarray with bounds check (Plan 09-02 / 09-03) defends. Documented as a Plan 09-02 contract: NAK retransmit handler MUST clamp seq to current file's frame count. Phase 9 SC#5 test does not exercise hostile seq values; Phase 12 fuzz target is the appropriate venue. |
