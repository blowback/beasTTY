// BestialiTTY Phase 3 — glyph atlas with OffscreenCanvas-backed tiles.
//
// Source: RESEARCH §Pattern 4 (Atlas class) + §Pattern 2 (bitmap rasteriser)
//         + §Pattern 3 (vector rasteriser) + CONTEXT D-03 (key + evict policy)
//         + PATTERNS.md §"www/renderer/atlas.js" (per-task verbatim shapes)
//
// Key shape: (ch << 24) | (fg << 16) | (nonce << 8) | zoom
//   - ch:   7-bit printable ASCII (0x20..0x7E); upper bits zero.
//   - fg:   8-bit palette index (for VT52 with no colour, always a fixed value —
//           use a single nonzero byte per theme variant to keep keys distinct
//           across phosphor changes even if nonce increment is skipped by bug).
//   - nonce: 8-bit atlas generation — incremented on every evict().
//   - zoom:  8-bit integer 1..4 (D-10).
//
// Evict on: theme change, phosphor change, zoom change, DPR change.
// Not LRU — full flush; worst-case cache is ~4 MB at 4× zoom + 2× DPR (trivial).

import { BITMAP_FONT } from './bitmap-font.js';

export class Atlas {
    constructor() {
        this.cache = new Map();
        this.nonce = 0;  // byte-wrapped; increments per evict
    }

    get(ch, fg, rasteriser, zoom) {
        const key = (ch << 24) | ((fg & 0xFF) << 16) | ((this.nonce & 0xFF) << 8) | (zoom & 0xFF);
        let tile = this.cache.get(key);
        if (!tile) {
            tile = rasteriser(ch, fg);   // Plan 02 wires this as a closure over theme + dpr
            this.cache.set(key, tile);
        }
        return tile;
    }

    evict() {
        this.cache.clear();
        this.nonce = (this.nonce + 1) & 0xFF;
    }

    size() {
        return this.cache.size;
    }
}

// Bitmap rasteriser (RESEARCH §Pattern 2, PATTERNS.md verbatim).
// Returns an OffscreenCanvas tile sized cellW*dpr × cellH*dpr, with the glyph
// for codepoint `ch` drawn in `fgColor` over a `bgColor` fill.
export function rasteriseBitmap(ch, fgColor, bgColor, cellW, cellH, z, dpr) {
    const tile = new OffscreenCanvas(cellW * dpr, cellH * dpr);
    const ctx = tile.getContext('2d');
    ctx.imageSmoothingEnabled = false;                // pixel-perfect CRT
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);            // NEVER ctx.scale (RESEARCH Anti-Pattern)
    ctx.fillStyle = bgColor;
    ctx.fillRect(0, 0, cellW, cellH);
    ctx.fillStyle = fgColor;
    const base = (ch & 0x7F) * 16;                      // 16 bytes per glyph; mask to ASCII
    for (let row = 0; row < 16; row++) {
        const bits = BITMAP_FONT[base + row];
        for (let col = 0; col < 8; col++) {
            if (bits & (0x80 >> col)) {
                ctx.fillRect(col * z, row * z, z, z);
            }
        }
    }
    return tile;
}

// Vector rasteriser (RESEARCH §Pattern 3, PATTERNS.md verbatim).
// MUST run AFTER document.fonts.ready — canvas.js gates this via bootRenderer.
// `fontPx` is the CSS font-size in pixels; the tile's pixel dimensions are
// cellW*dpr × cellH*dpr.
export function rasteriseVector(ch, fgColor, bgColor, cellW, cellH, fontPx, dpr) {
    const tile = new OffscreenCanvas(cellW * dpr, cellH * dpr);
    const ctx = tile.getContext('2d');
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);            // NEVER ctx.scale
    ctx.fillStyle = bgColor;
    ctx.fillRect(0, 0, cellW, cellH);
    ctx.font = `${fontPx}px "JetBrains Mono", ui-monospace, SFMono-Regular, Menlo, Consolas, monospace`;
    ctx.textBaseline = 'top';
    ctx.fillStyle = fgColor;
    ctx.fillText(String.fromCharCode(ch), 0, 0);
    return tile;
}

// Primer (RESEARCH §Pitfall 1): after atlas.evict(), force first drawImage
// for every printable ASCII so subsequent rAF ticks don't pay the GPU-texture
// upload cost. Call from canvas.js inside queueMicrotask(() => { ... }) after
// evict().
export function primeAscii(atlas, fg, rasteriser, zoom) {
    for (let ch = 0x20; ch <= 0x7E; ch++) {
        atlas.get(ch, fg, rasteriser, zoom);
    }
}

    dirtyView = new Uint8Array(wasm.memory.buffer, term.dirty_ptr(), term.rows());

function reDeriveViews() {

let canvas = null;

let atlas = null;

let activeZoom = DEFAULT_ZOOM;                    // D-06 default: 1

let rafPending = false;
let needsPaint = false;
let canvasHasFocus = false;
let hasPendingBell = false;

// ---- HiDPI resize (RESEARCH §Pattern 1) ----

function resizeToTheme() {
    const z = activeZoom;
    const cellW = activeTheme.cellW * z;
    const cellH = activeTheme.cellH * z;
    const cssW = cellW * 80;
    const cssH = cellH * 24;
    activeDpr = window.devicePixelRatio || 1;

    canvas.width  = Math.round(cssW * activeDpr);
    canvas.height = Math.round(cssH * activeDpr);
    canvas.style.width  = cssW + 'px';
    canvas.style.height = cssH + 'px';

    // ctx state is reset by canvas.width/height assignment (RESEARCH Pitfall #2).
    // Re-establish:
    ctx.setTransform(activeDpr, 0, 0, activeDpr, 0, 0);  // NEVER ctx.scale (Anti-Pattern)
    ctx.imageSmoothingEnabled = (activeTheme.name === 'crt' ? false : true);
    ctx.textBaseline = 'top';
    if (activeTheme.font) {
        ctx.font = `${activeTheme.fontPx}px "${activeTheme.font}", ui-monospace, monospace`;
    }

    // Paint the theme bg over the entire canvas so any non-dirty cells that
    // persisted from the previous theme vanish.
    ctx.fillStyle = activeTheme.bg;
    ctx.fillRect(0, 0, cssW, cssH);
}

// DPR change detection (monitor drag) — RESEARCH §Pattern 1 watchDPR.
function watchDPR() {
    const mql = window.matchMedia(`(resolution: ${window.devicePixelRatio}dppx)`);
    mql.addEventListener('change', () => {
        if (atlas) atlas.evict();
        resizeToTheme();
        needsPaint = true;
        requestFrame();
        watchDPR();   // re-register for the new DPR
    }, { once: true });

// ---- Atlas helpers ----

function makeRasteriserForTheme(theme) {
    const z = activeZoom;
    const dpr = activeDpr;
    if (theme.rasteriser === 'bitmap') {
        return (ch, _fg) => rasteriseBitmap(ch, theme.fg, theme.bg, theme.cellW * z, theme.cellH * z, z, dpr);
    } else {
        return (ch, _fg) => rasteriseVector(ch, theme.fg, theme.bg, theme.cellW * z, theme.cellH * z, theme.fontPx * z, dpr);
    }
}

function initAtlas() {
    atlas = new Atlas();
    const rast = makeRasteriserForTheme(activeTheme);
    // Prime in a microtask so first rAF paint isn't GPU-upload-blocked.
    queueMicrotask(() => primeAscii(atlas, /*fg=*/1, rast, activeZoom));
}

// ---- Paint ----

function paintRow(r, cols) {
    const z = activeZoom;
    const cellW = activeTheme.cellW * z;
    const cellH = activeTheme.cellH * z;
    const rast = makeRasteriserForTheme(activeTheme);

    // Clear the row band first (row bg) so dirty repaint overwrites old content.
    ctx.fillStyle = activeTheme.bg;
    ctx.fillRect(0, r * cellH, 80 * cellW, cellH);

    for (let c = 0; c < cols; c++) {
        const i = (r * cols + c) * CELL_SIZE;
        // ch is u32 LE — for printable ASCII only the low byte matters.
        let ch = gridView[i];
        if (ch === 0 || ch < 0x20) ch = 0x20;   // Blank non-printables as space (D-01 bitmap).
        const fg = 1;                             // VT52 is monochrome; single-fg palette index.
        const tile = atlas.get(ch, fg, rast, z);
        ctx.drawImage(tile, c * cellW, r * cellH, cellW, cellH);
    }
}

function paintCursor() {
    const z = activeZoom;
    const cellW = activeTheme.cellW * z;
    const cellH = activeTheme.cellH * z;
    const packed = term.cursor_packed();
    const row = packed >>> 16;
    const col = packed & 0xFFFF;
    const x = col * cellW;
    const y = row * cellH;

    if (!canvasHasFocus) {
        // Blurred: steady 1 px outlined block (D-07 / UI-SPEC Cursor table).
        ctx.strokeStyle = activeTheme.cursor.fgColor;
        ctx.lineWidth = 1;
        ctx.strokeRect(x + 0.5, y + 0.5, cellW - 1, cellH - 1);
        return;
    }

    // Focused: 530 ms blink via frameCount (D-07). 60 fps -> 32 frames ≈ 530 ms.
    const blinkOn = (frameCount % 64) < 32;
    if (!blinkOn) return;

    // Focused-on: paint cursor block + overdraw glyph in bgColor (inverted).
    ctx.fillStyle = activeTheme.cursor.fgColor;
    ctx.fillRect(x, y, cellW, cellH);

    const i = (row * term.cols() + col) * CELL_SIZE;
    let ch = gridView[i];
    if (ch === 0 || ch < 0x20) ch = 0x20;
    // Build an inverted tile on the fly (not atlased — only 1 cell per frame).
    // For CRT theme the inverted rasterisation swaps fg↔bg; for clean theme the
    // cursor fg is the accent color with glyph in the theme bg.
    const t = activeTheme;
    const invTile = (t.rasteriser === 'bitmap')
        ? rasteriseBitmap(ch, t.cursor.bgColor, t.cursor.fgColor, t.cellW * z, t.cellH * z, z, activeDpr)
        : rasteriseVector(ch, t.cursor.bgColor, t.cursor.fgColor, t.cellW * z, t.cellH * z, t.fontPx * z, activeDpr);
    ctx.drawImage(invTile, x, y, cellW, cellH);
}

// ---- rAF tick (RESEARCH §Complete rAF Loop Skeleton — verbatim shape) ----

function tick() {
    rafPending = false;
    frameCount++;
    reDeriveViews();

    // Bell (RESEARCH §Pattern 7 + CONTEXT D-08/D-09).
    if (term.bell_pending()) {
        triggerBellFlash();
        if (document.hidden && !hasPendingBell) {
            document.title = '(!) ' + ORIGINAL_TITLE;
            hasPendingBell = true;
        }
        term.clear_bell();
    }

    // Dirty-row repaint.
    term.snapshot_grid();
    const rows = term.rows();
    const cols = term.cols();
    for (let r = 0; r < rows; r++) {
        if (dirtyView[r] !== 0) paintRow(r, cols);
    }
    term.clear_dirty();

    // Cursor as overdraw (always — cheap at 80×24).
    paintCursor();

    // Self-reschedule (D-07 + RESEARCH Pitfall #4).
    if (canvasHasFocus || needsPaint) {
        needsPaint = false;
        requestFrame();
    }
}

export function requestFrame() {
    if (!rafPending) {
        rafPending = true;
        requestAnimationFrame(tick);
    }
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| wasm.memory.buffer → Uint8Array view | Grid bytes authored by Rust parser (trusted) flow into renderer as typed-array reads. No user-controlled path from DOM to grid mutation in this plan. |
| Codepoint from grid → drawn glyph | Non-printable bytes (< 0x20) are coerced to space (0x20) before atlas lookup. High-bit ASCII is masked via `(ch & 0x7F)` in rasteriseBitmap. |
| Theme descriptor mutation (phosphor) | Caller-supplied `color` string indexes `THEMES.crt.phosphorSlots[color]` — must validate against the enum. |
| DOM element lookup | `document.getElementById('terminal')` / `'bell-overlay'` — fails gracefully if Plan 03's DOM rewrite hasn't shipped. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-02-01 | Tampering | `setPhosphor(color)` unvalidated string | mitigate | `applyPhosphorToTheme` uses `if (!(color in THEMES.crt.phosphorSlots)) return;` before mutating. Enum values are `'green' \| 'amber' \| 'white'`. A caller passing `'__proto__'` finds no matching slot and the function returns early — no prototype pollution path. |
| T-03-02-02 | Tampering | `setTheme(name)` unvalidated string | mitigate | `if (!(name in THEMES)) return;` early-exits. Same defence as T-03-02-01. |
| T-03-02-03 | Information Disclosure | `document.title` prefix mutation | accept | The title prefix is `'(!) '` followed by the original title (captured once at module load). No user-controlled content in the title. |
| T-03-02-04 | Denial of Service | Atlas growth via pathological zoom-spam | mitigate | `zoomStep` clamps to [1, 4]. Atlas `evict()` fully clears on every zoom change. Worst-case single-atlas size (4× zoom + 2× DPR + 95 ASCII glyphs) is ~4 MB per RESEARCH §Pattern 4 — tolerable. GC reclamation is lazy (RESEARCH Pitfall #6) but bounded. |
| T-03-02-05 | Denial of Service | rAF pile-up via multi-queued `requestFrame()` | mitigate | `rafPending` guard prevents multiple outstanding rAF callbacks. `tick()` sets `rafPending = false` at entry before any work. |
| T-03-02-06 | Tampering | Inverted cursor rasterisation bypasses atlas cache | accept | Cursor inversion calls `rasteriseBitmap` / `rasteriseVector` on every focused-blink-on frame. Allocates one OffscreenCanvas per frame — measurable cost but not a DoS (one tile/16 ms ≈ 60 tiles/s, each ~1 KB). Acceptable for v1; a follow-up could cache inverted tiles keyed by `(ch, cursor_nonce)` if profiling shows jank. |
| T-03-02-07 | Tampering | XSS via DOM text write | mitigate | `canvas.js` writes to `document.title` (a property, not HTML-parsed), uses `.classList.add/remove` on overlay elements (no HTML sink), logs via `console.log` / `console.warn` only. No `innerHTML`, no `outerHTML`, no `insertAdjacentHTML`, no `document.write`. Acceptance criteria verify `grep -c "innerHTML" canvas.js` returns `0`. |
| T-03-02-08 | Information Disclosure | Font fingerprinting via `document.fonts.check` | accept | `document.fonts.check('14px "JetBrains Mono"')` returns true/false to console.log only — no cross-origin disclosure. Same-origin knowledge of which fonts loaded is not a privacy-sensitive surface. |
