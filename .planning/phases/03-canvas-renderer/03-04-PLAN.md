### File 1: `www/tests/render/grid.spec.js`
RENDER-01, RENDER-04, RENDER-05 — 80×24 grid renders fixture bytes.

```javascript
import { test, expect } from '@playwright/test';
import fs from 'node:fs';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const FIXTURE = fs.readFileSync(path.join(__dirname, '../fixtures/vt52-sample.bin'));

test.describe('RENDER-01 / RENDER-04 / RENDER-05 — 80×24 grid renders', () => {
  test('default CRT green paints fixture bytes with non-bg pixels', async ({ page }) => {
    await page.goto('/');
    // Wait for canvas to be sized (resizeToTheme) — any non-zero canvas.width will do.
    await page.waitForFunction(() => {
      const c = document.getElementById('terminal');
      return c && c.width > 0 && c.height > 0;
    });

    // Feed the fixture bytes via the Debug Feed button.
    const hexString = Array.from(FIXTURE).map(b => '\\x' + b.toString(16).padStart(2, '0').toUpperCase()).join('');
    await page.locator('#debug').evaluate(el => el.open = true);
    await page.fill('#input', hexString);
    await page.click('#feed');

    // Give the renderer one rAF tick to paint.
    await page.waitForTimeout(100);

    // Verify at least one cell in the top row is non-background.
    const nonBgFound = await page.evaluate(() => {
      const c = document.getElementById('terminal');
      const ctx = c.getContext('2d');
      const img = ctx.getImageData(0, 0, Math.min(c.width, 1280), 32);  // top row strip
      // Expect CRT green-phosphor bg (#0a0f0a ≈ rgb(10,15,10)) dominates; find any non-bg pixel.
      for (let i = 0; i < img.data.length; i += 4) {
        const r = img.data[i], g = img.data[i+1], b = img.data[i+2];
        // Not bg if red > 30 OR green > 50 (phosphor fg is #33ff66 ≈ rgb(51,255,102)).
        if (r > 30 || g > 50) return true;
      }
      return false;
    });
    expect(nonBgFound).toBe(true);
  });

  test('default CRT canvas matches visual baseline', async ({ page }) => {
    await page.goto('/');
    await page.waitForFunction(() => document.getElementById('terminal').width > 0);
    await page.waitForTimeout(200);   // let font-ready + initial paint settle
    // Mask the cursor region so blink phase doesn't cause flake.
    await expect(page.locator('#terminal-wrapper')).toHaveScreenshot('crt-default.png', {
      maxDiffPixelRatio: 0.02,
      mask: [page.locator('#terminal-wrapper').locator('//div[1]')],  // best-effort
    });
  });
});
```

```javascript
import { test, expect } from '@playwright/test';

test.describe('RENDER-10 — HiDPI backing store', () => {
  test('canvas.width equals cssWidth × devicePixelRatio @fast', async ({ page }) => {
    await page.goto('/');
    await page.waitForFunction(() => document.getElementById('terminal').width > 0);

    const { backingW, cssW, dpr } = await page.evaluate(() => {
      const c = document.getElementById('terminal');
      return {
        backingW: c.width,
        cssW: parseFloat(c.style.width) || c.clientWidth,
        dpr: window.devicePixelRatio,
      };
    });

    // Playwright projects=chromium uses deviceScaleFactor=2 per playwright.config.js.
    expect(dpr).toBe(2);
    expect(backingW).toBe(Math.round(cssW * dpr));
  });
});
```

### File 3: `www/tests/render/cursor.spec.js`
RENDER-02, RENDER-12 — block cursor visible + per-theme styling.

```javascript
import { test, expect } from '@playwright/test';

test.describe('RENDER-02 / RENDER-12 — Cursor', () => {
  test('focused cursor at (0,0) paints a visible block in CRT theme', async ({ page }) => {
    await page.goto('/');
    await page.waitForFunction(() => document.getElementById('terminal').width > 0);
    // Wrapper auto-focuses at boot (chrome.js) — wait one blink half-cycle.
    await page.waitForTimeout(300);

    // Read the top-left cell's centre pixel.
    const { r, g, b } = await page.evaluate(() => {
      const c = document.getElementById('terminal');
      const ctx = c.getContext('2d');
      // Cell is 16×32 CSS px at 1× CRT, 32×64 backing px at DPR=2.
      const img = ctx.getImageData(16, 32, 1, 1);   // middle of (0,0) cell
      return { r: img.data[0], g: img.data[1], b: img.data[2] };
    });

    // Cursor-on should paint the phosphor fg (#33ff66 ≈ rgb(51,255,102)) or its inversion.
    // Accept any non-bg pixel (bg is rgb(10,15,10)).
    const isCursorVisible = (r > 30 || g > 50);
    expect(isCursorVisible).toBe(true);
  });

  test('blurred cursor draws outlined block (no filled interior)', async ({ page }) => {
    await page.goto('/');
    await page.waitForFunction(() => document.getElementById('terminal').width > 0);

    // Blur the wrapper by clicking on Debug textarea.
    await page.locator('#debug').evaluate(el => el.open = true);
    await page.click('#input');

    await page.waitForTimeout(200);

    // Interior of the (0,0) cell should be near-bg; only the 1 px outline is fg.
    const { rInner, gInner, bInner } = await page.evaluate(() => {
      const c = document.getElementById('terminal');
      const ctx = c.getContext('2d');
      // Middle of cell — interior should be bg when blurred.
      const img = ctx.getImageData(16, 32, 1, 1);
      return { rInner: img.data[0], gInner: img.data[1], bInner: img.data[2] };
    });

    // Phosphor-bg is rgb(10,15,10); interior of blurred cursor should match closely.
    expect(rInner).toBeLessThan(60);
    expect(gInner).toBeLessThan(60);
    expect(bInner).toBeLessThan(60);
  });
});
```

```javascript
import { test, expect } from '@playwright/test';

const PALETTE = {
  green: '#33ff66',
  amber: '#ffb000',
  white: '#e8e8d8',
};

test.describe('RENDER-08 — Phosphor selection (CRT only)', () => {

    // Starting state: green pressed, others not.
    await expect(page.locator('[data-phosphor="green"]')).toHaveAttribute('aria-pressed', 'true');

    await expect(page.locator('[data-phosphor="amber"]')).toHaveAttribute('aria-pressed', 'true');
    await expect(page.locator('[data-phosphor="green"]')).toHaveAttribute('aria-pressed', 'false');
    await expect(page.locator('[data-phosphor="white"]')).toHaveAttribute('aria-pressed', 'false');

    await page.click('[data-phosphor="white"]');
    await expect(page.locator('[data-phosphor="white"]')).toHaveAttribute('aria-pressed', 'true');
  });

  test('CSS var --phosphor-fg matches selected palette', async ({ page }) => {
    await page.goto('/');

    for (const [color, hex] of Object.entries(PALETTE)) {

    await page.goto('/');

    await expect(page.locator('#phosphor-group')).toBeHidden();

});
```

### File 6: `www/tests/render/zoom.spec.js`
RENDER-09 — integer zoom 1..4 via Ctrl+/-/0.

```javascript
import { test, expect } from '@playwright/test';

async function cssWidth(page) {
  return page.evaluate(() => {
    const c = document.getElementById('terminal');
    return parseFloat(c.style.width);
  });
}

test.describe('RENDER-09 — Integer zoom via Ctrl +/-/0', () => {
  test('Ctrl+Equal zooms in; Ctrl+Minus zooms out; Ctrl+Digit0 resets @fast', async ({ page }) => {
    await page.goto('/');
    await page.locator('#terminal-wrapper').focus();
    await page.waitForFunction(() => document.getElementById('terminal').width > 0);

    const base = await cssWidth(page);   // CRT 1× = 16 * 80 = 1280 CSS px

    await page.keyboard.press('Control+Equal');
    await page.waitForTimeout(50);
    const z2 = await cssWidth(page);
    expect(z2).toBe(base * 2);

    await page.keyboard.press('Control+Equal');
    await page.waitForTimeout(50);
    const z3 = await cssWidth(page);
    expect(z3).toBe(base * 3);

    await page.keyboard.press('Control+Minus');
    await page.waitForTimeout(50);
    const z2b = await cssWidth(page);
    expect(z2b).toBe(base * 2);

    await page.keyboard.press('Control+Digit0');
    await page.waitForTimeout(50);
    const reset = await cssWidth(page);
    expect(reset).toBe(base);
  });

  test('zoom clamps at 4× (Ctrl+Equal past max is a no-op)', async ({ page }) => {
    await page.goto('/');
    await page.locator('#terminal-wrapper').focus();
    await page.waitForFunction(() => document.getElementById('terminal').width > 0);
    const base = await cssWidth(page);

    for (let i = 0; i < 6; i++) await page.keyboard.press('Control+Equal');
    await page.waitForTimeout(50);
    const capped = await cssWidth(page);
    expect(capped).toBe(base * 4);
  });
});
```
