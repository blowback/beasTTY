```rust
//! CORE-02 automated verification: the Rust core must have zero bindings to
//! Web Serial or any browser I/O API.
//!
//! Two checks, each a separate #[test]:
//!
//! 1. Dependency graph: `cargo metadata` must not list web-sys, js-sys, or
//!    wasm-bindgen in the resolved dep tree for the bestialitty-core package.
//!
//! 2. Source filesystem: no file under `crates/bestialitty-core/src/` may
//!    contain the strings `wasm_bindgen`, `web_sys`, or `js_sys`. (Plan 05
//!    removes the spike module; Phase 2 will add wasm_bindgen attrs to
//!    `lib.rs`, which is when this test will need to relax — but not in
//!    Phase 1.)
//!
//! Running `cargo test -p bestialitty-core --test core_02_no_browser_deps`
//! is the authoritative CORE-02 gate.

use std::path::PathBuf;
use std::process::Command;

#[test]
fn dependency_graph_excludes_browser_crates() {
    let output = Command::new(env!("CARGO"))
        .args(["metadata", "--format-version=1", "-p", "bestialitty-core", "--no-deps"])
        .output()
        .expect("cargo metadata should succeed");

    // --no-deps returns only the target package's direct manifest; we then
    // also fetch the full resolved tree for transitive verification.
    let full = Command::new(env!("CARGO"))
        .args(["metadata", "--format-version=1"])
        .output()
        .expect("cargo metadata (full) should succeed");

    let stdout = String::from_utf8_lossy(&full.stdout);

    for forbidden in ["\"name\":\"web-sys\"", "\"name\":\"js-sys\"", "\"name\":\"wasm-bindgen\""] {
        assert!(
            !stdout.contains(forbidden),
            "CORE-02 breach: cargo metadata contains {} — a browser dep has leaked into the bestialitty-core dep graph. \
             Production builds of the Rust core must not pull in Web Serial / DOM crates. \
             If Phase 2's lib.rs wasm-bindgen surface is being added, update this test to allow wasm-bindgen \
             BEHIND a feature gate rather than lifting the assertion outright.",
            forbidden
        );
    }

    // Smoke assertion that the package itself is in the metadata (guards
    // against a command failure silently passing this test).
    assert!(
        output.status.success() && stdout.contains("\"name\":\"bestialitty-core\""),
        "cargo metadata did not include bestialitty-core in its output"
    );
}

#[test]
fn source_files_contain_no_wasm_attrs() {
    let src_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("src");

    let forbidden_tokens = ["wasm_bindgen", "web_sys", "js_sys"];

    // Iterate every .rs file under src/ recursively.
    fn walk(dir: &std::path::Path, into: &mut Vec<PathBuf>) {
        for entry in std::fs::read_dir(dir).unwrap() {
            let entry = entry.unwrap();
            let path = entry.path();
            if path.is_dir() {
                walk(&path, into);
            } else if path.extension().and_then(|e| e.to_str()) == Some("rs") {
                into.push(path);
            }
        }
    }
    let mut files = Vec::new();
    walk(&src_dir, &mut files);
    assert!(!files.is_empty(), "no .rs files found under src/ — test is misconfigured");

    for path in &files {
        let contents = std::fs::read_to_string(path)
            .unwrap_or_else(|e| panic!("could not read {}: {}", path.display(), e));
        for token in &forbidden_tokens {
            assert!(
                !contents.contains(token),
                "CORE-02 breach: {} contains `{}`. Phase 1 logic modules must be wasm-free per D-20. \
                 Phase 2 adds wasm-bindgen attrs to lib.rs ONLY; when it does, update this test to \
                 ignore the lib.rs boundary file.",
                path.display(),
                token
            );
        }
    }
}

#[test]
fn crate_builds_for_native_target() {
    // Executing `cargo build --target x86_64-unknown-linux-gnu` from within a
    // test would be circular (we're already building); instead, we assert the
    // `[lib]` manifest key includes `rlib` by inspecting Cargo.toml directly.
    let cargo_toml = std::fs::read_to_string(
        PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("Cargo.toml")
    ).unwrap();

    assert!(
        cargo_toml.contains("crate-type = [\"cdylib\", \"rlib\"]")
         || cargo_toml.contains("crate-type = [\"rlib\", \"cdylib\"]"),
        "Cargo.toml must declare both cdylib and rlib crate types (D-19). \
         Current content: {}",
        cargo_toml
    );
}
```

```rust
//! Boundary API shape contract (D-17).
//!
//! Phase 2's wasm-bindgen wrapper in `lib.rs` will expose the Terminal's
//! public surface to JS. If any of these signatures drift, Phase 2 breaks
//! silently — the compiler will give a cryptic error at the wasm-pack
//! build. Asserting the shape here, as a compiling #[test], catches drift
//! the moment Phase 1 plans are edited.
//!
//! Every "test" below is a compile-time assertion in the guise of a runtime
//! fn call. If the signature changes, this file fails to compile, which is
//! the intended failure mode.

use bestialitty_core::terminal::Terminal;
use bestialitty_core::key::{encode, KeyCode, KeyEvent, Modifiers};

#[test]
fn terminal_constructor_signature_is_stable() {
    // D-12 / D-17: `Terminal::new(rows: u32, cols: u32, scrollback_cap: usize)`
    let _term: Terminal = Terminal::new(24u32, 80u32, 10_000usize);
}

#[test]
fn terminal_feed_accepts_byte_slice_returns_vec_u8() {
    // D-14 / D-17: `feed(&[u8]) -> Vec<u8>`
    let mut term = Terminal::new(24, 80, 100);
    let bytes: &[u8] = b"\x1BZ";          // ESC Z -> identify
    let reply: Vec<u8> = term.feed(bytes);
    assert_eq!(reply, vec![0x1B, b'/', b'K'],
        "ESC Z must return the canonical identify reply");
}

#[test]
fn terminal_state_queries_have_stable_return_types() {
    let term = Terminal::new(24, 80, 100);
    let _: (u32, u32) = term.cursor();    // D-17 `cursor()` -> (row, col)
    let _: u32 = term.rows();
    let _: u32 = term.cols();
    let _: bool = term.bell_pending();
    let _: &[u8] = term.dirty();
}

#[test]
fn terminal_mutations_have_stable_signatures() {
    let mut term = Terminal::new(24, 80, 100);
    term.clear_bell();
    term.clear_dirty();
    term.resize(30u32, 90u32);
    term.resize_scrollback(5_000usize);
}

#[test]
fn key_encode_signature_is_stable() {
    // Phase 4 will call this with a packed KeyEvent struct.
    let evt = KeyEvent {
        code: KeyCode::ArrowUp,
        mods: Modifiers::default(),
    };
    let bytes: Vec<u8> = encode(evt);
    assert_eq!(bytes, vec![0x1B, b'A']);
}

#[test]
fn key_event_constructors_exist() {
    // Phase 4 convenience constructors.
    let _a = KeyEvent::new(KeyCode::Enter);
    let _b = KeyEvent::with_ctrl(KeyCode::Char(b'c'));
    let _ = Modifiers::NONE;
    let _ = Modifiers::CTRL;
}

#[test]
fn terminal_exposes_grid_read_access() {
    // Renderer in Phase 3 reads grid via Terminal::grid() -> &Scrollback.
    // Scrollback's public API (row, visible_rows, cols) is verified here
    // indirectly by being callable.
    let term = Terminal::new(24, 80, 100);
    let sb = term.grid();
    assert_eq!(sb.visible_rows(), 24);
    assert_eq!(sb.cols(), 80);
}

#[test]
fn esc_y_end_to_end_through_feed() {
    // Integration smoke: ESC Y decoded via feed() produces the correct cursor.
    let mut term = Terminal::new(24, 80, 100);
    term.feed(b"\x1BY\x37\x6F"); // bottom-right
    assert_eq!(term.cursor(), (23, 79));
}

#[test]
fn feed_across_two_calls_is_stateful() {
    // Core correctness of a byte-streaming parser — split ESC Y across feed() calls.
    let mut term = Terminal::new(24, 80, 100);
    term.feed(b"\x1BY");
    term.feed(b"\x23\x45");
    assert_eq!(term.cursor(), (3, 37));
}

#[test]
fn bell_end_to_end() {
    let mut term = Terminal::new(24, 80, 100);
    assert!(!term.bell_pending());
    term.feed(b"\x07");
    assert!(term.bell_pending());
    term.clear_bell();
    assert!(!term.bell_pending());
}
```

fail to compile if `pub` visibility is ever accidentally narrowed to
`pub(crate)` on a D-17 method, or if a return type drifts. Phase 2's
wasm-bindgen wrapper will consume exactly this surface.
